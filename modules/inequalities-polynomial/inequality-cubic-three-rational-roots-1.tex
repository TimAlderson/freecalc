$
\begin{array}{rclll}
\alertNoH{3,4}{\alertNoH{7}{ 2} x^{3}-5 x^{2}+x+2}  &> &0\\
\uncover<3->{\fcAnswerUncover{3}{7}{2}\left(x-\fcAnswer{5}{\left(\alertNoH{8}{-\frac{1}{2}}\right)} \right) (x-\fcAnswer{5}{\alertNoH{9}{ 1}})(x-\fcAnswer{5}{\alertNoH{10}{2}}) &> &0}\\
\uncover<17->{\alertNoH{17,18}{x\in}\fcAnswer{18}{ \left(-\infty, -\frac{1}{2}\right)\cup \left( 1,2\right)} }
\end{array}
$

\uncover<8->{Left hand side vanishes when $\alertNoH{8}{x=-\frac{1}{2}}$, when $\alertNoH{9}{x=1}$ and when $\alertNoH{10}{x=2}$. } \uncover<11->{The two roots split the real line into four intervals: $ \alertNoH{11}{\left(-\infty,-\frac{1}{2} \right)}, \alertNoH{12}{ \left(-\frac{1}{2},1\right)}, \alertNoH{13}{ \left(1,2 \right)} , \alertNoH{14}{\left(2,\infty \right)}$. }
