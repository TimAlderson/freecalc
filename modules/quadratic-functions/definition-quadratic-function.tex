\begin{frame}
\begin{definition}
Let $a,b,c$ be real numbers with $a\neq 0$. The function 
\[
f(x)=ax^2+bx+c
\]
is called a \emph{quadratic function}.
\end{definition}
\begin{itemize}
\item<2-> The graph of a quadratic function is called a parabola.
\end{itemize}
