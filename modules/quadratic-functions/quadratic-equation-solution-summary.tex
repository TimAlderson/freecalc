\column{0.7\textwidth}
The solutions of the quadratic equation 
\[
ax^2+bx+c=0
\]
are the numbers 
\[
x_1=\frac{-b+ \alertNoH{3}{\sqrt{\alertNoH{2}{b^2-4ac}}}}{2a} \qquad x_2=\frac{-b- \alertNoH{3}{\sqrt{\alertNoH{2}{b^2-4ac}}}}{2a}.
\]

\end{columns}
\end{theorem}
\begin{itemize}
\item<2-> Abbreviated as 
\[
x_1,x_2=\frac{-b\alertNoH{4}{\pm \alertNoH{3}{\sqrt{\alertNoH{2}{b^2-4ac}}}} }{2a}=\frac{-b\alertNoH{4}{ \pm \alertNoH{3}{\sqrt{ \alertNoH{2}{D}} }}}{2a}, \qquad \text{ where }\alertNoH{2}{D} =\alertNoH{2}{b^2-4ac}.
\]
\item<3-> If $\alertNoH{3}{D < 0}$ then $\alertNoH{3}{\sqrt{ D} }$ is not a real $\Rightarrow$ quadratic has no real solutions.
\item<4-> If $\alertNoH{4}{D=0}$ then $x_1=x_2$, the equation has only one zero (with multiplicity two). \uncover<5->{The zero is located at the vertex of the parabola.}
\end{itemize}
