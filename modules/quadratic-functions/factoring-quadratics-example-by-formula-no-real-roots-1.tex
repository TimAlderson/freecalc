\begin{frame}
\hfil\hfil $ax^2+bx+c=a\left( x- x_1\right)\left(x-x_2\right),\quad\quad\quad\quad \left|\begin{array}{rcl}
x_1x_2&=&\displaystyle \frac{c}{a}\\
x_1+x_2&=&\displaystyle-\frac{b}{a}
\end{array}\right.
$
