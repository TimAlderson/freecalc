\begin{example}
Find $\Arccos (\cos 4)$.
\begin{itemize}
\item<2-> $4$ is not between $0$ and $\pi$.
\item<3-> We need the angle \alertNoH{8,12}{$a$ between $0$ and $\pi$} for which $\alertNoH{11}{ \alertNoH{4}{ \cos 4} =\alertNoH{7}{ \cos a}}$.
\end{itemize}
\uncover<9->{
\[
\begin{array}{rcl}
\alertNoH{9,10,13}{a} &\alertNoH{9,10,14}{=}& \fcAnswer{10}{ \alertNoH{11}{\pi -} (4\alertNoH{11}{-\pi})}\uncover<11->{=\alertNoH{14}{\alertNoH{11}{2\pi}-4}} \\
\uncover<12->{\text{Therefore }\quad \Arccos(\alertNoH{12}{\sin 4}) &\alertNoH{0}{ =} & \alertNoH{12}{\Arccos(\alertNoH{12}{\sin a})}} \\
\uncover<13->{&\alertNoH{0}{ =}& \worksheet{ \alertNoH{13,14}{ a} \uncover<14->{\alertNoH{14}{= 2\pi-4}.}}}
\end{array}
\]
}
\end{example}
