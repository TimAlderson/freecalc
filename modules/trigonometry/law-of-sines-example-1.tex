A triangle has a \alertNoH{2}{side of length $2cm$}; the two angles adjacent to it are $30^\circ$ and $45^\circ$. 
\begin{itemize}
\item Find the other two sides of the triangle.
\item Find the area of the triangle.
\end{itemize}
\end{columns}
\only<handout:1|1-7>{
\begin{itemize}
\item<2-> \alertNoH{2}{Let the known side be $c=2cm$.}
\item<3-> Let the known angles \alertNoH{3}{$30^\circ$, $45^{\circ}$ be arranged as in the figure}\uncover<4->{, and let the \alertNoH{4}{third angle be $ \alertNoH{5,6}{\gamma} \uncover<5->{\alertNoH{5,6}{=} \fcAnswer{6}{180^\circ -30^\circ- 45^\circ= 180^\circ -75^\circ = 105^\circ}} $}.}
\item<7-> Label the unknown sides $\alertNoH{7}{a,b}$ as indicated.
\end{itemize}
}
