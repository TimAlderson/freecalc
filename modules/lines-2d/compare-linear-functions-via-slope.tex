% begin module compare-linear
\begin{frame}
\psset{xunit=0.6cm, yunit=0.6cm}
\begin{pspicture}(-1,-4)(5.5,4.3)
\tiny
\fcAxesStandard{-1}{-4}{5}{4}
\psline[linecolor=red](-1,-1)(4,4)
\rput[lt](3,3){$y=x$}
\uncover<2->{%
\psline[linecolor=black](-1,-2)(2,4)%
\rput[lt](1.8,3.6){$y=2x$}%
}%
\uncover<3->{%
\psline[linecolor=blue](-1,-3)(! 4 3 div 4)%
\rput[rt](1.3,3.9){$y=3x~$}%
}%
\uncover<4->{%
\psline[linecolor=black](-1,-0.5)(! 5 2.5)%
\rput[br](3,1.5){$y=\frac{x}{2}$}%
}%
\uncover<5->{%
\psline[linecolor=blue](!-1 -1 3 div)(! 5 5 3 div)%
\rput[lt](!4 4 3 div){$y=\frac{x}{3}$}%
}%
\fcLabels{5}{4}
\fcXTickWithLabel{1}{$1$}
\fcYTickWithLabel{1}{$1$}
\end{pspicture}

\begin{itemize}
\item<1->  If two linear functions have positive slopes, the one with the bigger slope increases faster.
\item<2->  $y = 2x$ increases twice as fast as $y = x$.
\item<3->  $y = 3x$ increases three times as fast as $y = x$.
\item<4->  $y = \frac{x}{2}$ increases half as fast as $y = x$.
\item<5->  $y = \frac{x}{3}$ increases one third as fast as $y = x$.
\end{itemize}
\end{frame}

\begin{frame}

\tiny

\psline[linecolor=red](-1,1)(4,-4)
\rput[lb](3,-3){$y=-x$}
\uncover<2->{%
\psline[linecolor=black](-1,2)(2,-4)%
\rput[lb](1.8,-3.6){$y=-2x$}%
}%
\uncover<3->{%
\psline[linecolor=blue](-1,3)(! 4 3 div -4)%
\rput[rb](1.3,-3.9){$y=-3x~$}%
}%
\uncover<4->{%
\psline[linecolor=black](-1, 0.5)(! 5 -2.5)%
\rput[tr](3,-1.5){$y=-\frac{x}{2}$}%
}%
\uncover<5->{%
\psline[linecolor=blue](!-1 1 3 div)(! 5 5 -3 div)%
\rput[lb](!4 4 -3 div){$y=-\frac{x}{3}$}%
}%
\fcLabels{5}{4}
\fcXTickWithLabel{1}{$1$}
\fcYTickWithLabel{1}{$1$}
\end{pspicture}
\begin{itemize}
\item<1->  If two linear functions have negative slopes, the one with the lower slope decreases faster.
\item<2->  $y = -2x$ decreases twice as fast as $y = -x$.
\item<3->  $y = -3x$ decreases three times as fast as $y = -x$.
\item<4->  $y = -\frac{1}{2}x$ decreases half as fast as $y = -x$.
\item<5->  $y = -\frac{1}{3}x$ decreases one third as fast as $y = -x$.
\end{itemize}
\end{frame}
